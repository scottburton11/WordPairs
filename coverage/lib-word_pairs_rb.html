30   # Run the matcher, outputting word pairs to stdout.
 
31   def self.run(words = WordPairs::Words, dictionary = WordPairs::Dictionary, trim = 2)
 
32     words.each do |word|
 
33       begin
 
34         word.trim = trim
 
35         result = ReverseAlternateFinder.find(word.alternate, dictionary) if word.has_alternate?
 
36         puts "#{word}, #{result[0]}" unless result.nil?
 
37       rescue Exception => e
 
38         puts "There was an error with #{word}, the exception raised was: #{e}"
 
39         break
 
40       end
 
41     end
 
42   end
